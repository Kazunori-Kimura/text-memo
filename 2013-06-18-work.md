
create table cp_data (
    drcode      char(6),
    drname      varchar2(200),
    ncc_cd      char(7),
    ncc_name    varchar2(300),
    ncc_dept    varchar2(300),
    title       varchar2(100),
    category    varchar2(30),
    kingaku     number(14, 0),
    kaisu       number(3, 0)
);




select
    d.*,
    m.cont_id,
    m.doccd,
    m.lst_nm || 'Å@' || m.fst_nm as m_dr_name,
    m.name as m_name,
    m.ktkn_nm as m_kana
from
    cp_data d,
    coa_doc1p m
where
    d.drcode = '008752' and
    m.doccd = d.drcode;




select
    d.*,
    hsp.act_uni_id,
    hsp.acnt_nm,
    hsp.acnt_jpln_nm,
    hsp.acnt_ktkn_nm
from
    cp_data d,
    coa_dcfhsp hsp
where
    d.ncc_cd = '2600027' and
    d.ncc_cd = hsp.hs01 || hsp.hs02
